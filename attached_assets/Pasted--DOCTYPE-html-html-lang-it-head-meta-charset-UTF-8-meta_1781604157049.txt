function buildHeatmap() {
  const hm = $('heatmap');
  if (hm.children.length > 0) return;
  const cols = ['transparent','rgba(124,107,255,0.12)','rgba(124,107,255,0.3)','rgba(124,107,255,0.55)','rgba(124,107,255,0.78)','#7c6bff'];
  for (let i = 0; i < 52 * 5; i++) {
    const r = Math.random();
    const ci = r < 0.25 ? 0 : r < 0.45 ? 1 : r < 0.62 ? 2 : r < 0.78 ? 3 : r < 0.90 ? 4 : 5;
    const d = document.createElement('div');
    d.className = 'hm-cell';
    d.style.background = cols[ci];
    if (ci > 0) d.style.border = '1px solid rgba(124,107,255,0.15)';
    hm.appendChild(d);
  }
}

const SYSTEM_PROMPT = `Sei un assistente AI integrato nella piattaforma Semantic Growth Index (SGI).
Rispondi in modo approfondito, stimolante e interdisciplinare nella lingua dell'utente.
Dopo ogni risposta aggiungi ESATTAMENTE questo blocco su una riga separata (niente altro attorno):
SGI_UPDATE:{"complessita":X,"varieta":X,"interdiscipl":X,"profondita":X,"originalita":X,"probabilistico":X,"sintesi":X,"stabilita":X,"delta":X}
Dove X per le 8 dimensioni = interi 0-100 che riflettono la qualità del messaggio dell'utente.
"delta" = float 0.0-1.5: messaggi semplici 0.0-0.2, messaggi profondi/interdisciplinari 0.5-1.5.
Non affermare mai di misurare intelligenza, QI, personalità o salute mentale.`;

async function callAI(userMsg) {
  STATE.chatHistory.push({ role: 'user', content: userMsg });

  /* 
    In produzione su Replit usa /api/chat (server-side).
    Questo chiamata diretta funziona solo in development con proxy.
  */
  const res = await fetch('https://api.anthropic.com/v1/messages', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      model: 'claude-sonnet-4-6',
      max_tokens: 1000,
      system: SYSTEM_PROMPT,
      messages: STATE.chatHistory.slice(-20)
    })
  });

  if (!res.ok) throw new Error('API ' + res.status);

  const data = await res.json();
  const fullText = data.content.map(b => b.text || '').join('');
  const match = fullText.match(/SGI_UPDATE:(\{[^}]+\})/);
  const cleanText = fullText.replace(/\nSGI_UPDATE:\{[^}]+\}/, '').trim();
  let update = null;
  if (match) { try { update = JSON.parse(match[1]); } catch(e) {} }

  STATE.chatHistory.push({ role: 'assistant', content: fullText });
  return { text: cleanText, update };
}

function applyUpdate(update) {
  if (!update) return;
  const delta = parseFloat(update.delta) || 0;
  STATE.sgi = Math.min(100, Math.round((STATE.sgi + delta) * 10) / 10);
  STATE.sessionDelta = Math.round((STATE.sessionDelta + delta) * 10) / 10;
  STATE.xp += Math.round(delta * 40 + 10);

  const dimMap = {
    complessita: 'Complessità', varieta: 'Varietà semantica',
    interdiscipl: 'Interdiscipl.', profondita: 'Profondità',
    originalita: 'Originalità', probabilistico: 'Probabilistico',
    sintesi: 'Sintesi', stabilita: 'Stabilità'
  };
  Object.entries(dimMap).forEach(([k, label]) => {
    if (update[k] !== undefined)
      STATE.dims[label] = Math.min(100, Math.round(STATE.dims[label] * 0.75 + update[k] * 0.25));
  });

  updateUI();

  // TODO: BACKEND — salva aggiornamento SGI:
  // await fetch('/api/sessions', { method:'POST', body: JSON.stringify({ delta, dims: update }) });
}

function addMessage(text, isUser, update) {
  const msgs = $('chat-messages');
  const div = document.createElement('div');
  div.className = 'msg ' + (isUser ? 'user' : 'ai');

  const av = document.createElement('div');
  av.className = 'msg-avatar';
  av.textContent = isUser ? STATE.user.initials : 'AI';

  const body = document.createElement('div');
  body.className = 'msg-body';

  const bubble = document.createElement('div');
  bubble.className = 'msg-bubble';
  bubble.textContent = text;
  body.appendChild(bubble);

  if (!isUser && update && parseFloat(update.delta) > 0) {
    const d = parseFloat(update.delta);
    const tags = [];
    if (update.interdiscipl > 72) tags.push(STATE.lang === 'es' ? 'conexiones interdiscipl.' : STATE.lang === 'en' ? 'interdisciplinary' : 'connessioni interdiscipl.');
    if (update.complessita > 72) tags.push(STATE.lang === 'es' ? 'alta complejidad' : STATE.lang === 'en' ? 'high complexity' : 'alta complessità');
    if (update.originalita > 72) tags.push(STATE.lang === 'es' ? 'originalidad' : STATE.lang === 'en' ? 'originality' : 'originalità');

    const upd = document.createElement('div');
    upd.className = 'sgi-update';
    upd.innerHTML = `<i class="ti ti-trending-up" style="font-size:11px"></i> SGI → <strong>${STATE.sgi.toFixed(1)}</strong> <span class="pos">+${d.toFixed(1)}</span>${tags.length ? ' ' + tags.map(tg => `<span class="sgi-tag">${tg}</span>`).join('') : ''}`;
    body.appendChild(upd);
  }

  div.appendChild(av);
  div.appendChild(body);
  msgs.appendChild(div);
  msgs.scrollTop = msgs.scrollHeight;
}

function showTyping() {
  const msgs = $('chat-messages');
  const div = document.createElement('div');
  div.className = 'msg ai'; div.id = 'typing-ind';
  div.innerHTML = '<div class="msg-avatar">AI</div><div class="msg-body"><div class="typing-indicator"><span></span><span></span><span></span></div></div>';
  msgs.appendChild(div);
  msgs.scrollTop = msgs.scrollHeight;
}
function hideTyping() { const t = $('typing-ind'); if (t) t.remove(); }

async function sendMessage() {
  const inp = $('chat-input');
  const btn = $('send-btn');
  const text = inp.value.trim();
  if (!text || btn.disabled) return;
  inp.value = ''; inp.style.height = 'auto';
  STATE.msgCount++;
  addMessage(text, true, null);
  btn.disabled = true;
  showTyping();

  try {
    const { text: reply, update } = await callAI(text);
    hideTyping();
    applyUpdate(update);
    addMessage(reply, false, update);
  } catch (err) {
    hideTyping();
    const errText = STATE.lang === 'en' ? 'Connection error. Please try again.'
      : STATE.lang === 'es' ? 'Error de conexión. Inténtalo de nuevo.'
      : 'Errore di connessione. Riprova tra qualche istante.';
    addMessage(errText, false, null);
  }
  btn.disabled = false;
}

$('send-btn').addEventListener('click', sendMessage);
$('chat-input').addEventListener('keydown', e => {
  if (e.key === 'Enter' && !e.shiftKey) { e.preventDefault(); sendMessage(); }
});
$('chat-input').addEventListener('input', function() {
  this.style.height = 'auto';
  this.style.height = Math.min(this.scrollHeight, 110) + 'px';
});
